// (Seu código login.js - não precisa de alterações) document.addEventListener('DOMContentLoaded', () => { const loginForm = document.getElementById('loginForm'); if (loginForm) { loginForm.addEventListener('submit', async (event) => { event.preventDefault(); const login = document.getElementById('loginInput').value; const password = document.getElementById('passwordInput').value; try { const response = await fetch('http://localhost:3000/login', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ login, password }) }); const data = await response.json(); if (response.ok) { alert(data.message); window.location.href = 'dashboard.html'; // Redireciona para o dashboard após login bem-sucedido } else { alert(data.error); } } catch (error) { console.error('Erro de rede ou servidor:', error); alert('Erro ao tentar fazer login. Verifique sua conexão ou o servidor.'); } }); } });
